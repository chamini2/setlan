Leyenda:
S:		Comienzo
Y:		Instruccion o Bloque
I:		Instruccion
B:		Comienzo del bloque
D:		Declaracion de variables
IB:		Instruccion de bloque
S:		Strings
C:		Condicional
EI:		else if y else
F:		for
DIR:	min o max para recorrer el set
W:		repeat while
A:		Asignacion
E:		Expresiones
PC: 	Coma y Punto y coma
TIP:	tipos
TKS:	variables o constantes
NUM:	numeros
BOL:	booleanos
SET:	sets
NS:		numeros del set



S 	->	program Y
Y 	->	I | B
I 	->	println S | print S | scan tokID | if C Y | for F | W | A 
B 	->	{ using D in IB } | { IB }
D 	-> 	TKS tokID ; | TKS tokID , D
IB 	->	println S ; | print S ; | scan tokID ; | if C Y EI ;| A ; | W ;
S 	->	tokString | E | tokString , S | E , S
EI 	->	else if C Y EI | else Y
C 	->	( E )
F 	->	tokID DIR SET do Y
DIR -> 	min | max
W 	->	repeat Y while C do  Y | while C do  Y | repeat Y while C 
A 	->	tokID = E
E 	-> 	TKS | - TKS | ( E ) | TKS E' | >? E | <? E | $? E | not E 
E' 	-> 	+ E | - E | * E | / E | % E |
		++ E | \ E | >< E | 
		<+> E | <-> E | <*> E | </> E | <%> E | 
		== E | /= E | or E | and E | < E | <= E | > E | >= E
TIP	->	int | bool | set
TKS ->	NUM | BOL | SET
NUM ->	tokID | tokNum
BOL ->	tokID | tokTrue | tokFalse
SET -> 	tokID | { NS }
NS 	->  NUM | NUM , NS
